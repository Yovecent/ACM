#include <bits/stdc++.h>

using namespace std;
using i64 = long long;
const int maxn = 2e6 + 10;
const int mod = 1e9 + 7;

i64 f[maxn],inv[maxn];
struct info{

    i64 val;
    int x,y;
    info(){}
    info(i64 val,int x,int y):val(val),x(x),y(y){};
    friend bool operator < (info x,info y){
        if(x.x == y.x) return x.y < y.y;
        return x.x < y.x;
    }
};
vector<info>v;
i64 ksm(i64 a,int b){

    i64 res = 1;
    while(b){
        if(b&1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}
i64 ok(int x,int y,int xt,int yt){
    
    x = xt - x ;
    y = yt - y ;

    return f[x + y] * inv[x] % mod * inv[y] % mod;
}
